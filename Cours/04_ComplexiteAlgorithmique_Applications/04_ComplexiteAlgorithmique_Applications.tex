\question{Donner la complexité temporelle, dans le pire des cas et dans le meilleur des cas, de l'algorithme suivant.}
\begin{lstlisting}
def f(L):
    x = 0
    for i in range(len(L)) :
        for j in range(len(L[0])) :
            x = x+1
    return x
\end{lstlisting}
\ifprof
$\mathcal{O}(n\times m)$
\else
\fi



\question{Donner la complexité temporelle, dans le pire des cas et dans le meilleur des cas, de l'algorithme suivant.}
\begin{lstlisting}
def f2(n) :
    x = 0
    for i in range (n) :
        for j in range (i) :
            x += 1
    return x
\end{lstlisting}
\ifprof

\else
\fi



\question{Donner la complexité temporelle, dans le pire des cas et dans le meilleur des cas, de l'algorithme suivant.}
\begin{lstlisting}
def f3(n) :

    while i > 1:
        x += 1
        i //= 2
    return x
\end{lstlisting}

\fi
