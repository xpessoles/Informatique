On considère la fonction suivante. 
%\begin{Verbatim}[gobble=0,numbers=left]
\begin{lstlisting}

def mystere(a,b) :
    """Précondition : a,b sont des entiers, a>0, b>1"""
    k,p = 0,1
    while a % p == 0 :
        k = k+1
        p = p*b
    return k-1
\end{lstlisting}
    
%\end{Verbatim}
