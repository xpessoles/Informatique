\question{On considère les affectations \texttt{a} $\leftarrow$ \texttt{-1.} et \texttt{b} $\leftarrow$ \texttt{5}.
Prévoir la valeur de chacune de ces expressions, puis le vérifier à l'aide de l'interpréteur interactif.}

\begin{multicols}{2}
  \begin{enumerate}[label=\emph{\alph*)}]
    \item \texttt{a * a}
    \item \texttt{a ** a}
    \item \texttt{a ** a == a}
    \item \texttt{a * b}
    \item \texttt{a+b == 5}
    \item \texttt{a+6>=b}
    \item \texttt{b<100 | a**2==-1}
    \item \texttt{b-3}
  \end{enumerate}
\end{multicols}
