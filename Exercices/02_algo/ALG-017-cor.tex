\question{Proposer un invariant de boucle pour démontrer cet algorithme.}

\begin{pyverbatim}
def est_premier(n):
    """ Renvoie True si n est premier, False sinon
        Préconditon : n est un entier."""
    for d in range(2,n):
        # n n'est pas divisible par 2, 3, ..., d-1.
        if n % d == 0:
            return False
    return True
\end{pyverbatim}
