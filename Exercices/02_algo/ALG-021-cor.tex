\question{Soit les algorithmes de recherche de maximum d'un tableau ci-dessous, proposez des invariants de boucles.}
\begin{lstlisting}
def maxi(t):
    """Renvoie le plus grand élément de t.
       Précondition : t est un tableau 
       non vide"""
    m = t[0]
    for x in t:
        # Invariant : m est le plus grand 
        # élément trouvé jusqu'ici
        if x > m:
            m = x # On a trouvé plus grand, 
                  # on met à jour m
    return m
\end{lstlisting}

def maxi(t):
    """Renvoie le plus grand élément de t.
       Précondition : t est un tableau 
       non vide"""
    m = t[0] # Initialisation par le 
             # premier élément
    for i in range(1, len(t)):
        # Invariant : m == max(t[0:i])
        if t[i] > m:
            m = t[i] # On a trouvé plus
                     # grand, on met à 
                     # jour m
    return m

\question{Soit les algorithmes de recherche d'indice de maximum d'un tableau ci-dessous, proposez des invariants de boucles.}

\begin{lstlisting}
def indicemaxi(t):
    """Renvoie l'indice du plus grand 
        élément de t.
       Précondition : t est un tableau 
       non vide"""
    im = 0 # Indice du maximum, 
           # initialisation par 
           # le premier élément
    for i in range(1, len(t)):
        # Invariant : im est indice d'un 
        # plus grand élément de t[0:i]
        if t[i] > t[im]:
            im = i # On a trouvé plus grand, 
                   # on met à jour im
    return im
    
\end{lstlisting}

def indicemaxi(t):
    """Renvoie l'indice du plus grand élément 
       de t.
       Précondition : t est un tableau 
       non vide"""
    im = 0 # Indice du maximum, 
            # initialisation par le

        # Invariant : im est indice 
        # d'un plus grand élément de t[0:i]

            im = i # On a trouvé plus grand,
