p20 = pyramide(a,20)

def total_pyramide(p) :
    """Calcule la somme maximale dans la pyramide p"""
    n = len(p)
    m = 0
    for ligne in range(n) :
        for j in range(n-ligne-1) :
            p[n-ligne-2][j] += max(p[n-ligne-1][j],p[n-ligne-1][j+1])
    return p[0][0]
