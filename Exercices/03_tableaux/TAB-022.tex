On considère la fonction suivante.
\begin{Verbatim}[gobble=0,numbers=left]
def mystere(n):
    """Préconditions : n entier positif"""
    L = []
    c = 0 
    while c ** 2 <= n : 
        L.append(c**2)
        c = c+1
    return L
\end{Verbatim}
