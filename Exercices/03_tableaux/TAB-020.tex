On considère la fonction suivante.
\begin{Verbatim}[gobble=0,numbers=left]
def suite(n):
    """Précondition : n entier naturel"""
    u = 2
    for i in range(n) : 
        u = 4 * u - 3 
    return u
\end{Verbatim}
