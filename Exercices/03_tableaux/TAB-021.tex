On considère la fonction suivante.
\begin{lstlisting}%[gobble=0,numbers=left]
def mystere(L,M):
    """Préconditions : L tableau de nombres, M nombre"""
    S = 0
    n = len(L)
    for i in range(n) : 
        if L[i] > M : 
            S = S + L[i]
    return S
\end{lstlisting}
