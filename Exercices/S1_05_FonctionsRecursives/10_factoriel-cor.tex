
\question{Proposer une fonction \texttt{fact$\_$it(n:int)->int} renvoyant la factorielle de $n$ par une méthode itérative}


\begin{lstlisting}
def fact_it(n : int) -> int :
    res = 1
    for i in range (1,n+1): # ou range(n)
        res = res*i
    return res
\end{lstlisting}


\question{Proposer une fonction \texttt{fact$\_$rec(n:int)->int} renvoyant la factorielle de $n$ par une méthode récursive}


\begin{lstlisting}
def fact_rec(n : int) -> int :
    if n == 0 : 
        return 1
    else : 
        return n * fact_rec(n-1) # Récursion
\end{lstlisting}
